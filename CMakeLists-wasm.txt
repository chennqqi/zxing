cmake_minimum_required(VERSION 3.10)
project(zxing-wasm)

# 设置Emscripten工具链
set(CMAKE_TOOLCHAIN_FILE $ENV{EMSDK}/upstream/emscripten/cmake/Modules/Platform/Emscripten.cmake)

# 设置C++标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 设置输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# WASM 平台不支持动态链接，强制使用静态库
set(BUILD_SHARED_LIBS OFF CACHE BOOL "Build and link as shared library" FORCE)

# 禁用 ZXingCPP 的示例、测试和 Python 绑定，避免不必要的依赖（如 stb, Qt, OpenCV）
set(ZXING_EXAMPLES OFF CACHE BOOL "Build the example barcode reader/writer applications" FORCE)
set(ZXING_BLACKBOX_TESTS OFF CACHE BOOL "Build the black box reader/writer tests" FORCE)
set(ZXING_UNIT_TESTS OFF CACHE BOOL "Build the unit tests" FORCE)
set(ZXING_INSTALL_TEST OFF CACHE BOOL "Test if the installed/public API works" FORCE)
set(ZXING_PYTHON_MODULE OFF CACHE BOOL "Build the python module" FORCE)

# Emscripten特定设置
# STANDALONE_WASM=1 produces a WASM that doesn't depend on Emscripten JS runtime
# This allows wazero (pure Go WASM runtime) to load and execute the module
set(EMSCRIPTEN_LINK_FLAGS
    -s WASM=1
    -s STANDALONE_WASM=1
    -s EXPORTED_FUNCTIONS='["_malloc","_free","_create_default_options","_free_options","_decode_barcode","_decode_barcode_data","_decode_barcodes","_decode_result_get","_free_result","_free_results","_get_last_error"]'
    -s ALLOW_MEMORY_GROWTH=1
    -O3
    -Wl,--no-entry
)

# 添加ZXingCPP子目录
add_subdirectory(zxing-cpp)

# 创建WASM wrapper库
add_executable(zxingwrapper src/zxing.cpp)

# 设置包含目录
target_include_directories(zxingwrapper
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/include
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/src
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src/oned
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src/qrcode
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src/datamatrix
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src/aztec
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src/pdf417
        ${CMAKE_CURRENT_SOURCE_DIR}/zxing-cpp/core/src/maxicode
)

# 链接ZXing库
target_link_libraries(zxingwrapper
    PRIVATE
        ZXing
)

# 设置Emscripten链接标志
set_target_properties(zxingwrapper PROPERTIES
    LINK_FLAGS "${EMSCRIPTEN_LINK_FLAGS}"
    SUFFIX ".wasm"
)

# 安装规则
install(TARGETS zxingwrapper
    RUNTIME DESTINATION bin
)

install(FILES include/zxing.h
    DESTINATION include
)
